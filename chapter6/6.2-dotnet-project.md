## 核心实现

### 配置服务

```csharp
// Program.cs
var builder = WebApplication.CreateBuilder(args);

// 添加 AI 服务
builder.Services.AddOpenAIChatClient(
    new Uri("https://api.openai.com/v1"),
    "your-api-key");

// 添加 Semantic Kernel
builder.Services.AddSingleton(sp =>
    Kernel.CreateBuilder()
        .AddOpenAIChatCompletion("gpt-4", "your-api-key")
        .Build());
```

### Chat 服务

```csharp
public class ChatService
{
    private readonly IChatClient _chatClient;

    public ChatService(IChatClient chatClient)
    {
        _chatClient = chatClient;
    }

    public async Task<string> ChatAsync(string message)
    {
        var response = await _chatClient.GetResponseAsync(
            "你是一个有帮助的助手。",
            message);

        return response.Message.Text;
    }
}
```

### API 控制器

```csharp
[ApiController]
[Route("api/[controller]")]
public class ChatController : ControllerBase
{
    private readonly ChatService _chatService;

    public ChatController(ChatService chatService)
    {
        _chatService = chatService;
    }

    [HttpPost]
    public async Task<IActionResult> Post([FromBody] ChatRequest request)
    {
        var response = await _chatService.ChatAsync(request.Message);
        return Ok(new ChatResponse { Message = response });
    }
}
```

## 工具调用

### 定义工具

```csharp
public class WeatherPlugin
{
    [KernelFunction]
    [Description("获取指定城市的天气信息")]
    public string GetWeather(
        [Description("城市名称")] string city)
    {
        // 实际应用中调用天气 API
        return $"city: {city}, weather: sunny, temp: 25°C";
    }
}
```

### 注册工具

```csharp
var kernel = Kernel.CreateBuilder()
    .AddOpenAIChatCompletion("gpt-4", "your-api-key")
    .Build();

kernel.Plugins.AddFromType<WeatherPlugin>("Weather");
```

### 自动调用

```csharp
var settings = new OpenAIPromptExecutionSettings
{
    ToolCallBehavior = ToolCallBehavior.AutoInvokeKernelFunctions
};

var response = await _chatClient.GetResponseAsync(
    systemMessage: "你是一个天气助手。",
    userMessage: "北京的天气怎么样？",
    executionSettings: settings,
    kernel: kernel);
```

## 会话管理

### 对话历史

```csharp
public class Conversation
{
    public string Id { get; set; }
    public List<ChatMessage> Messages { get; } = new();
}

public class ChatMessage
{
    public string Role { get; set; }  // user/assistant/system
    public string Content { get; set; }
    public DateTime Timestamp { get; set; }
}
```

### 管理服务

```csharp
public class ConversationService
{
    private readonly Dictionary<string, Conversation> _conversations = new();

    public Conversation GetOrCreate(string id)
    {
        if (!_conversations.ContainsKey(id))
            _conversations[id] = new Conversation { Id = id };
        return _conversations[id];
    }

    public void AddMessage(string conversationId, string role, string content)
    {
        var conv = GetOrCreate(conversationId);
        conv.Messages.Add(new ChatMessage
        {
            Role = role,
            Content = content,
            Timestamp = DateTime.UtcNow
        });
    }
}
```

## 流式响应

```csharp
[HttpGet("stream")]
public async Task Stream(string message)
{
    Response.ContentType = "text/event-stream";

    await foreach (var chunk in _chatService.StreamChatAsync(message))
    {
        await Response.WriteAsync($"data: {chunk}\n\n");
        await Response.Body.FlushAsync();
    }
}
```
