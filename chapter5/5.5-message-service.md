# 5.5 消息服务与对话 API

## 消息数据模型

```python
from pydantic import BaseModel
from typing import Optional, Literal
from datetime import datetime

class Message(BaseModel):
    id: str
    session_id: str
    role: Literal["user", "assistant", "system"]
    content: str
    created_at: datetime = datetime.now()
    metadata: dict = {}
```

## 消息服务实现

```python
class MessageServiceImpl(MessageService):
    def __init__(self, db):
        self.db = db
        self.messages = {}  # session_id -> [messages]
    
    async def add_message(self, session_id: str, message_data: dict) -> Message:
        message = Message(
            id=str(uuid.uuid4()),
            session_id=session_id,
            role=message_data["role"],
            content=message_data["content"],
            metadata=message_data.get("metadata", {})
        )
        
        if session_id not in self.messages:
            self.messages[session_id] = []
        self.messages[session_id].append(message)
        
        return message
    
    async def get_messages(
        self,
        session_id: str,
        limit: int = 50,
        before: Optional[str] = None
    ) -> list[Message]:
        messages = self.messages.get(session_id, [])
        
        if before:
            messages = [m for m in messages if m.id != before]
        
        return messages[-limit:]
    
    async def get_message(self, message_id: str) -> Optional[Message]:
        for messages in self.messages.values():
            for msg in messages:
                if msg.id == message_id:
                    return msg
        return None
    
    async def update_message(self, message_id: str, content: str) -> Message:
        message = await self.get_message(message_id)
        if message:
            message.content = content
        return message
    
    async def delete_message(self, message_id: str) -> bool:
        for session_id, messages in self.messages.items():
            for i, msg in enumerate(messages):
                if msg.id == message_id:
                    del messages[i]
                    return True
        return False
    
    async def clear_messages(self, session_id: str) -> bool:
        if session_id in self.messages:
            self.messages[session_id] = []
            return True
        return False
    
    async def get_conversation_context(
        self,
        session_id: str,
        max_tokens: int = 4000
    ) -> list[dict]:
        messages = await self.get_messages(session_id, limit=100)
        
        context = []
        total_tokens = 0
        
        for message in reversed(messages):
            msg_tokens = len(message.content) // 4
            if total_tokens + msg_tokens > max_tokens:
                break
            context.insert(0, {
                "role": message.role,
                "content": message.content
            })
            total_tokens += msg_tokens
        
        return context
```

## 对话 API 集成

### 构建对话请求

```python
async def build_chat_request(
    session_id: str,
    user_input: str,
    role_service: RoleService,
    message_service: MessageService
) -> list[dict]:
    """构建发送给大模型的对话请求"""
    
    # 获取会话信息
    session = await session_service.get_session(session_id)
    if not session:
        raise ValueError(f"Session {session_id} not found")
    
    # 获取角色信息
    role = await role_service.get_role(session.role_id)
    if not role:
        raise ValueError(f"Role {session.role_id} not found")
    
    # 构建消息列表
    messages = [{"role": "system", "content": role.system_prompt}]
    
    # 添加历史上下文
    context = await message_service.get_conversation_context(session_id)
    messages.extend(context)
    
    # 添加当前用户输入
    messages.append({"role": "user", "content": user_input})
    
    return messages
```

### 处理对话响应

```python
async def process_chat_response(
    session_id: str,
    response: str,
    message_service: MessageService
) -> Message:
    """处理大模型响应，保存消息"""
    
    # 保存助手回复
    assistant_message = await message_service.add_message(
        session_id,
        {"role": "assistant", "content": response}
    )
    
    # 更新会话时间
    await session_service.update_session(session_id)
    
    return assistant_message
```
