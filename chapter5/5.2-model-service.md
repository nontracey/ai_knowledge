# 5.2 模型服务层实现

## 模型服务层设计

### 设计目标

- 封装不同模型提供商的调用逻辑
- 提供统一的模型调用接口
- 支持模型提供商的灵活切换

### 核心接口

```python
from abc import ABC, abstractmethod

class BaseModelProvider(ABC):
    """模型提供商抽象基类"""
    
    @abstractmethod
    async def chat(self, messages: list, **kwargs) -> str:
        """发送对话请求"""
        pass
    
    @abstractmethod
    async def stream_chat(self, messages: list, **kwargs):
        """流式对话请求"""
        pass
    
    @abstractmethod
    async def embeddings(self, texts: list) -> list:
        """获取文本嵌入向量"""
        pass
```

## 实现模型提供商服务类

### OpenAI 提供商

```python
from openai import AsyncOpenAI
from .base import BaseModelProvider

class OpenAIProvider(BaseModelProvider):
    def __init__(self, api_key: str, base_url: str = None, model: str = "gpt-4"):
        self.client = AsyncOpenAI(api_key=api_key, base_url=base_url)
        self.model = model
    
    async def chat(self, messages: list, **kwargs) -> str:
        response = await self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            **kwargs
        )
        return response.choices[0].message.content
    
    async def stream_chat(self, messages: list, **kwargs):
        stream = await self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            stream=True,
            **kwargs
        )
        async for chunk in stream:
            if chunk.choices[0].delta.content:
                yield chunk.choices[0].delta.content
    
    async def embeddings(self, texts: list) -> list:
        response = await self.client.embeddings.create(
            model="text-embedding-3-small",
            input=texts
        )
        return [item.embedding for item in response.data]
```

## 模型服务工厂

```python
from enum import Enum

class ModelType(Enum):
    OPENAI = "openai"
    DEEPSEEK = "deepseek"
    ZHIPU = "zhipu"

class ModelServiceFactory:
    _providers = {}
    
    @classmethod
    def register(cls, model_type: ModelType, provider: BaseModelProvider):
        cls._providers[model_type] = provider
    
    @classmethod
    def get_provider(cls, model_type: ModelType) -> BaseModelProvider:
        provider = cls._providers.get(model_type)
        if not provider:
            raise ValueError(f"Provider for {model_type} not registered")
        return provider
    
    @classmethod
    async def chat(cls, model_type: ModelType, messages: list, **kwargs) -> str:
        provider = cls.get_provider(model_type)
        return await provider.chat(messages, **kwargs)
```

## 使用示例

```python
# 注册提供商
factory = ModelServiceFactory()
factory.register(ModelType.OPENAI, OpenAIProvider(api_key="xxx"))
factory.register(ModelType.DEEPSEEK, DeepSeekProvider(api_key="xxx"))

# 调用模型
response = await factory.chat(
    ModelType.DEEPSEEK,
    messages=[{"role": "user", "content": "你好"}]
)
print(response)
```
