# 5.4 角色与会话管理

## 角色管理服务

### 角色数据模型

```python
from pydantic import BaseModel
from typing import Optional
from datetime import datetime

class Role(BaseModel):
    id: str
    name: str
    description: str
    system_prompt: str
    avatar: Optional[str] = None
    created_at: datetime = datetime.now()
    updated_at: datetime = datetime.now()
```

### 角色服务接口

```python
class RoleService:
    async def create_role(self, role_data: Role) -> Role:
        """创建新角色"""
        pass
    
    async def get_role(self, role_id: str) -> Optional[Role]:
        """获取角色信息"""
        pass
    
    async def update_role(self, role_id: str, role_data: Role) -> Role:
        """更新角色信息"""
        pass
    
    async def delete_role(self, role_id: str) -> bool:
        """删除角色"""
        pass
    
    async def list_roles(self) -> list[Role]:
        """列出所有角色"""
        pass
```

### 角色服务实现

```python
class RoleServiceImpl(RoleService):
    def __init__(self, db):
        self.db = db
        self.roles = {}
    
    async def create_role(self, role_data: Role) -> Role:
        role_data.id = str(uuid.uuid4())
        self.roles[role_data.id] = role_data
        await self.db.save("roles", role_data.dict())
        return role_data
    
    async def get_role(self, role_id: str) -> Optional[Role]:
        return self.roles.get(role_id)
    
    async def update_role(self, role_id: str, role_data: Role) -> Role:
        if role_id in self.roles:
            role_data.id = role_id
            role_data.updated_at = datetime.now()
            self.roles[role_id] = role_data
            return role_data
        raise ValueError(f"Role {role_id} not found")
    
    async def delete_role(self, role_id: str) -> bool:
        if role_id in self.roles:
            del self.roles[role_id]
            return True
        return False
    
    async def list_roles(self) -> list[Role]:
        return list(self.roles.values())
```

## 会话管理服务

### 会话数据模型

```python
class Session(BaseModel):
    id: str
    role_id: str
    user_id: str
    title: str
    created_at: datetime = datetime.now()
    updated_at: datetime = datetime.now()
    metadata: dict = {}
```

### 会话服务接口

```python
class SessionService:
    async def create_session(self, role_id: str, user_id: str) -> Session:
        """创建新会话"""
        pass
    
    async def get_session(self, session_id: str) -> Optional[Session]:
        """获取会话信息"""
        pass
    
    async def update_session(self, session_id: str, **kwargs) -> Session:
        """更新会话"""
        pass
    
    async def delete_session(self, session_id: str) -> bool:
        """删除会话"""
        pass
    
    async def list_user_sessions(self, user_id: str) -> list[Session]:
        """列出用户的所有会话"""
        pass
    
    async def add_message(self, session_id: str, message: dict) -> bool:
        """添加消息到会话"""
        pass
```

### 会话服务实现

```python
class SessionServiceImpl(SessionService):
    def __init__(self, db, message_service):
        self.db = db
        self.message_service = message_service
        self.sessions = {}
    
    async def create_session(self, role_id: str, user_id: str) -> Session:
        session = Session(
            id=str(uuid.uuid4()),
            role_id=role_id,
            user_id=user_id,
            title="新会话"
        )
        self.sessions[session.id] = session
        return session
    
    async def get_session(self, session_id: str) -> Optional[Session]:
        return self.sessions.get(session_id)
    
    async def update_session(self, session_id: str, **kwargs) -> Session:
        session = self.sessions.get(session_id)
        if session:
            for key, value in kwargs.items():
                setattr(session, key, value)
            session.updated_at = datetime.now()
        return session
    
    async def delete_session(self, session_id: str) -> bool:
        if session_id in self.sessions:
            del self.sessions[session_id]
            await self.message_service.clear_messages(session_id)
            return True
        return False
    
    async def list_user_sessions(self, user_id: str) -> list[Session]:
        return [s for s in self.sessions.values() if s.user_id == user_id]
    
    async def add_message(self, session_id: str, message: dict) -> bool:
        return await self.message_service.add_message(session_id, message)
```
