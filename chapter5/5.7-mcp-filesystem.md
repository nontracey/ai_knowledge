## 代码实现

### 初始化 MCP 客户端

```python
from mcp import Client
from mcp.config import Config

class MCPClientManager:
    def __init__(self):
        self.clients = {}
    
    async def initialize(self, config: Config):
        for server_name, server_config in config.servers.items():
            client = Client()
            await client.connect(server_config)
            self.clients[server_name] = client
    
    async def call_tool(self, server_name: str, tool_name: str, params: dict):
        client = self.clients.get(server_name)
        if not client:
            raise ValueError(f"Server {server_name} not found")
        return await client.call_tool(tool_name, params)
```

### 调用文件系统工具

async def read_file_content(client_manager: MCPClientManager, path: str):

    try:
        result = await client_manager.call_tool(

        return result

    try:
        result = await client_manager.call_tool(
            "filesystem",

### 处理返回结果

```python
async def process_file_operation(client_manager: MCPClientManager, operation: str, params: dict):
    """处理文件操作请求"""
    result = await client_manager.call_tool("filesystem", operation, params)
    
    # 处理不同类型的结果
    if isinstance(result, dict):
        if "error" in result:
            raise ValueError(result["error"])
        return result.get("content", result)
    
    return result
```
